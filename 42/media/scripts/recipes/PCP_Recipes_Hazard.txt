/* ==================================================================================== */
/* ============================== HEALTH HAZARD RECIPE VARIANTS ======================== */
/* ==================================================================================== */
/*  When SandboxVars.PCP.EnableHealthHazards is ON, the 10 original hazardous recipes
    are hidden (via modified OnTest) and replaced by these 20 Safe/Unsafe twins.

    SAFE (Protected):
      - Requires gas mask or respirator WITH filter + goggles + gloves
      - Filter degrades 0.025 per craft (~40 crafts to exhaust a standard filter)
      - No health effects

    UNSAFE (No Mask!):
      - Requires gloves only (no mask, no goggles)
      - Triggers EHR disease (if EHR mod active) or vanilla stat penalties (fallback)
      - Partial protection scaling if player happens to be wearing a mask              */

    craftRecipe PCPDistillMethanolSafe
    {

        category = PhobosChemistry,
